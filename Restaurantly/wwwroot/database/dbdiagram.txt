Table roles {
  id INT [pk, increment]
  role_name VARCHAR(50)
}

Table categories {
  id INT [pk, increment]
  name VARCHAR(255)
  sort_order INT
}

Table customers {
  id INT [pk, increment]
  full_name VARCHAR(255)
  email VARCHAR(255)
  phone VARCHAR(20)
  password VARCHAR(255)
  reset_token VARCHAR(255)
  reset_token_expiry DATETIME
  status BOOLEAN
  date_added DATETIME
}

Table tables {
  id INT [pk, increment]
  table_number INT
  seats INT
  status BOOLEAN
}

Table reservations {
  id INT [pk, increment]
  customer_id INT
  table_id INT
  number_of_guests INT
  reservation_time DATETIME
  status BOOLEAN
  date_added DATETIME
  date_modified DATETIME
}

Table products {
  id INT [pk, increment]
  name VARCHAR(255)
  category_id INT
  price DECIMAL(10, 2)
  description TEXT
  featured BOOLEAN
  image VARCHAR(255)
  status BOOLEAN
  date_added DATETIME
  date_modified DATETIME
}

Table orders {
  id INT [pk, increment]
  customer_id INT
  reservation_id INT
  total DECIMAL(10, 2)
  status BOOLEAN
  payment_status BOOLEAN
  date_added DATETIME
  date_modified DATETIME
}

Table order_items {
  id INT [pk, increment]
  order_id INT
  product_id INT
  quantity INT
  price DECIMAL(10, 2)
}

Table product_images {
  id INT [pk, increment]
  product_id INT
  image VARCHAR(255)
  sort_order INT
}

Table users {
  id INT [pk, increment]
  username VARCHAR(255)
  password VARCHAR(255)
  email VARCHAR(255)
  role_id INT
  status BOOLEAN
  created_at DATETIME
}

Table settings {
  id INT [pk, increment]
  key VARCHAR(255)
  value VARCHAR(255)
}

Table contacts {
  id INT [pk, increment]
  full_name VARCHAR(255)
  email VARCHAR(255)
  phone VARCHAR(20)
  subject VARCHAR(255)
  message TEXT
  date_added DATETIME
  date_modified DATETIME
  status BOOLEAN
}

Table banner_images {
  id INT [pk, increment]
  image VARCHAR(255)
  link VARCHAR(255)
  sort_order INT
  status BOOLEAN
}

Table category_paths {
  id INT [pk, increment]
  category_id INT
  path_id INT
  level INT
}

Table reviews {
  id INT [pk, increment]
  customer_id INT
  product_id INT
  rating INT
  comment TEXT
  date_added DATETIME
}

// Define Relationships
Ref: roles.id < users.role_id
Ref: categories.id < products.category_id
Ref: customers.id < reservations.customer_id
Ref: tables.id < reservations.table_id
Ref: customers.id < orders.customer_id
Ref: reservations.id < orders.reservation_id
Ref: orders.id < order_items.order_id
Ref: products.id < order_items.product_id
Ref: products.id < product_images.product_id
Ref: categories.id < category_paths.category_id
Ref: reviews.customer_id < customers.id
Ref: reviews.product_id < products.id
